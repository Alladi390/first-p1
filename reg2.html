import pandas as pd

def correlation(dataset, threshold):
    col_corr = set()  # To store the names of columns with high correlation
    corr_matrix = dataset.corr()  # Calculate the correlation matrix
    
    # Iterate over the correlation matrix
    for i in range(len(corr_matrix.columns)):
        for j in range(i):  # Only check the lower triangle of the matrix (i < j)
            if abs(corr_matrix.iloc[i, j]) > threshold:  # Check if correlation is higher than the threshold
                colname = corr_matrix.columns[i]  # Get column name
                col_corr.add(colname)  # Add to the set of highly correlated columns
                
    return col_corr
